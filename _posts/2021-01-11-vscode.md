# 工作区间

目录清单

```json
{
	"folders": [
		{
			"name": "bin",
			"path": "server\\bin"
    }
  ]
}
```

调试，附加进程

```json
{
	"launch": {
		"version": "0.2.0",
		"configurations": [
			{
				"name": "Attach to Process",
				"type": "go",
				"request": "attach",
				"mode": "local",
				// "remotePath": "${workspaceFolder:bin}\\Lobby.exe",
				"presentation": {
					"hidden": false,
					"group": "1",
					"order": 2
				},
				"processId": 17984
      },
      {
				"name": "Login.exe",
				"type": "go",
				"request": "launch",
				"mode": "exec",
				"program": "${workspaceFolder:bin}\\Login.exe",
				"env": {},
				"args": []
      }
    ]
  }
}
```

组合节点

```json
{
		"compounds": [
			{
				"name": "Build/RunAll",
				"configurations": [
					// "Login.exe",
					"Gateway.exe",
					"DBServer.exe",
					"Lobby.exe",
					"Scene.exe",
					"Matcher.exe",
					"Level.exe"
				],
				"presentation": {
					"hidden": false,
					"group": "1",
					"order": 1
				},
				"preLaunchTask": "buildall",
				"stopAll": true,
      }
    ]
}
```

任务节点

```json
{
	"tasks": {
		"version": "2.0.0",
		"tasks": [
			{
				"label": "buildall",
				"type": "shell",
				"command": "go",
				"args": [
					"build",
					"-gcflags",
					"'-N -l'",
					// "-race",
					"-o",
					"${workspaceFolder:bin}",
					"./..."
				],
				"options": {
					"cwd": "${workspaceFolder:src}"
				},
				"problemMatcher": [],
				"group": {
					"kind": "build",
					"isDefault": true
				}
      }
    ]
  }
}
```
